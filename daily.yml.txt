name: Daily Blog Generator
on:
  schedule:
    - cron: '30 3 * * *'
  workflow_dispatch: 

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: write  # This allows the bot to save the text file
    steps:
      - uses: actions/checkout@v4
      
      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'
          
      - name: Install dependencies
        run: pip install google-generativeai feedparser
          
      - name: Run script
        env:
          GEMINI_API_KEY: ${{ secrets.GEMINI_API_KEY }}
        run: python main.py

      - name: Commit and Push the Text File
        run: |
          git config --global user.name "BlogBot"
          git config --global user.email "bot@github.com"
          git add .
          git commit -m "Added daily blogs for $(date +'%Y-%m-%d')"
          git push
